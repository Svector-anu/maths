\chapter{Differential Equations}

% -------------------------------------------------------
\section{Ordinary Differential Equations}
% -------------------------------------------------------

\begin{definition}[Ordinary Differential Equation]
An \textbf{ordinary differential equation} (ODE) is an equation involving derivatives of a function of a \emph{single} independent variable.
\end{definition}

\begin{example}
\[
  \frac{dy}{dx} + y = x
\]
Here $y$ is a function of the single variable $x$.
\end{example}

% -------------------------------------------------------
\section{Partial Differential Equations}
% -------------------------------------------------------

\begin{definition}[Partial Differential Equation]
A \textbf{partial differential equation} (PDE) is an equation involving partial derivatives of a function of \emph{two or more} independent variables.
\end{definition}

\begin{example}
\[
  \frac{\partial u}{\partial x} + \frac{\partial u}{\partial y} = 0
\]
Here $u$ is a function of both $x$ and $y$.
\end{example}
